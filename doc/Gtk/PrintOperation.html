def run(action=:print_dialog, parent=nil, &blk)
	action = PrintOperation.const_get("ACTION_#{action}".upcase) 
	if blk
		run_(action, parent){|result| 
			blk.call self,result.nick.gsub(/-/, "_").to_sym
		}
	else
		run_(action, parent)
	end
end
