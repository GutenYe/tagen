def initialize(arg)
# arg: int or str		
	if arg.class == Fixnum
		@code = arg
		@key = @@codes[arg]
	elsif arg.class == String
		@key = arg
		@code = @@keys[arg]
	end
end

def <=>(arg)
	if arg.class == self.class
		code <=> arg.code
	elsif arg.class == Fixnum
		code <=> arg
	elsif arg.class == String
		code <=> @@keys[arg]
