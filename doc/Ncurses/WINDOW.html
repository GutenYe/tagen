def convert_xy x=nil,y=nil
	x = self.x+x.to_i if x and String===x
	y = self.y+y.to_i if y and String===y

	if x and y
		[x,y]
	elsif x
		x
	elsif y
		y
	end
end

def getc
	key = getch
	if key==27
	c = getch
	(value=KEYS[[27, c]]) ?  value : "\e"+c.chr
	elsif value=KEYS[key]
	value
	else
	# a "\C-a"
	key.chr
	end
end

def mvxy x,y, &blk
	x,y = convert_xy(x,y)
	x_, y_ = self.xy
	Ncurses.move(y,x) 
	if blk
	blk.call
	Ncurses.move(y_, x_)
	end
end
